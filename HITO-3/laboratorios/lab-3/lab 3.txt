create or replace function cadenas (a text , b2 text )
returns text
begin
    declare str text default '' ;

    if char_length (a, b2 ) and strcmp(a , b2) >  15

        then set str = ' son cadenas igulaes  ';
        else
        set str = ' cadenas distintas ';

    end if;
    return  str;

end;

returns text
begin

        set str = concat(' si existe ' ,LOCATE(extension,ci ));

end;
